Danh sách các chuỗi và tệp.

Thường thì một lập trình viên phải lưu và tải dữ liệu văn bản từ một tệp. Thư viện Qt để làm việc với tệp có lớp QFile, sẽ được thảo luận chi tiết hơn ở phần sau.Chúng ta sẽ làm quen với lớp bổ trợ QTextStream và xem xét mã tải văn bản từ một tệp.

Để làm việc với tệp và luồng văn bản,sẽ bao gồm các mô-đun tiêu đề bắt buộc:

#include <QFile>
#include <QTextStream>

Mã để tải văn bản từ một tệp vào QStringList sẽ trông giống như sau:

 // khai báo một biến - một danh sách các chuỗi     
    QStringList sl ;  
    // Khai báo một biến tệp và đặt tên tệp        
    QFile file( "c: \\ test.ini" );  
    // Đang cố gắng mở tệp   
    if ( file . open ( QFile :: ReadOnly | QFile :: Text )) {   
        // Tạo một luồng văn bản , chuyển cho nó một con trỏ tệp        
        QTextStream ts (& file ); 
        // Cho đến khi tệp kết thúc    
        while (! ts.atEnd ( ) ) { 
            // đọc dòng tiếp theo và đưa nó vào danh sách các dòng         
            sl << ts . readLine ();  
        }
        // Đóng tệp  
        file . close();
   }

Lưu kết quả vào một tệp có thể trông giống như sau:

 // Khai báo một biến tệp và đặt tên cho tệp đã lưu        
    QFile resFile ( "c: \\ Temp \\ result.txt" ); 
    // Đang cố gắng mở tệp để ghi     
    if ( resFile . open ( QFile :: WriteOnly | QFile :: Text )) {    
        // Tạo một luồng văn bản , chuyển cho nó một con trỏ tệp        
        QTextStream ts (& resFile ); 
        // Đối với mỗi dòng từ danh sách     
        for ( auto line: sl ) {   
            // đưa nó vào dòng văn bản và thêm "end of line"         
            ts << line << '\ n' ;    
        }
        // Đóng tệp  
        file . close ();
    }
