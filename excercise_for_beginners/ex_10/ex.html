Đề bài:

Viết chương trình tạo một chuỗi mới từ các ký tự số của một chuỗi cho trước theo thuật toán sau: mỗi chữ số được thay bằng (+) cho một chữ số chẵn và [-] cho một chữ số lẻ, trong khi số + hoặc - các ký tự phải bằng giá trị của chữ số.

Lưu ý:ký tự 0 được thay thế bằng chuỗi ().

Định dạng đầu vào:chuỗi ký tự.

Định dạng đầu ra:nội dung của dòng được tạo theo thuật toán được mô tả.

Ví dụ (dòng đầu tiên là đầu vào của chuỗi nguồn, dòng thứ hai là đầu ra của kết quả).

Code :

    1 #include <iostream>
    2 
    3 using namespace std;
    4 
    5 int main()
    6 {
    7     string s;
    8     getline(cin,s);
    9     string a ="";
   10     for(int i=0;i<s.size();i++){
   11         if((s[i] >= '0' && s[i] <= '9')){
   12             if(int(s[i] - 48) % 2 == 0){
   13                 a+='(';
   14                 for(int j = 0;j < int(s[i] - 48);j++){
   15                     a+='+';
   16                 }
   17                 a+=')';
   18             }
   19             else if(int(s[i] - 48) % 2 != 0){
   20                 a+='[';
   21                 for(int j = 0;j < int(s[i] - 48);j++){
   22                     a+='-';
   23                 }
   24                 a+=']';
   25             }
   26         }
   27     }
   28     cout<<a;
   29     return  0;
   30 }
