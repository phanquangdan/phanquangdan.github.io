Đề bài:

Viết chương trình đếm số từ trong một chuỗi đã cho. Một từ là một chuỗi ký tự tùy ý (chữ cái, số, dấu câu, v.v.) được phân tách bằng một hoặc nhiều dấu cách.

Định dạng đầu vào:chuỗi ký tự.

Định dạng đầu ra:số từ trong chuỗi đã cho.

Ví dụ (dòng đầu tiên là đầu vào của chuỗi nguồn, dòng thứ hai là đầu ra của kết quả).

Code :

    1 #include <iostream>
    2 
    3 using namespace std;
    4 
    5 void editS(string &s){
    6     while(s[0] == ' '){
    7         s.erase(0,1);
    8     }
    9     while(s[s.size()-1] == ' '){
   10         s.erase(s.size()-1,1);
   11     }
   12     while(s.find("  ") != -1){
   13         s.erase(s.find("  "),1);
   14     }
   15 }
   16 
   17 int main()
   18 {
   19     string s;
   20     int count = 1;
   21     getline(cin,s);
   22     editS(s);
   23     for(int i=0;i<s.size();i++){
   24         if(s[i] == ' '){
   25             count++;
   26         }
   27     }
   28     if(s.size() == 0) cout<<0;
   29     else cout<<count;
   30     return  0;
   31 }
